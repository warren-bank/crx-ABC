  var data            = window['__abc_com__']
  var is_episode_list = (data.page.type === 'show')  && (data.page.subType === 'episode-guide')
  var is_episode      = (data.page.type === 'video') && (data.page.subType === 'episode')
  var video, brand, show_id, video_type, video_id, locked, episode, page

  if (is_episode) {
    video      = data.page.content.video.layout.video
    brand      = video.brand || '001'
    show_id    = ((video.show instanceof Object) && video.show.id)
                  ? video.show.id
                  : '-1'
    video_type = video.type || 'lf'
    video_id   = video.id
    locked     = video.accesslevel && (video.accesslevel !== "0")
    episode    = {
      brand:       brand,
      show_id:     show_id,
      video_type:  video_type,
      video_id:    video_id,
      locked:      locked,
      title:       video.title       || '',
      description: video.description || '',
      duration_ms: video.duration    || 0,
      season:      video.seasonnumber,
      episode:     video.episodenumber
    }
  }

  page = {
    is_episode_list: is_episode_list,
    is_episode:      is_episode,
    episode:         episode || null
  }
